Usage of yuque-sync:
  -n   string
          设置namespace
  [-o] bool
          文件同名覆盖写入 (default true)
  [-p] string
          设置存储路径 (default "download/")
  [-t] string
          文件写入格式,支持hexo、hugo (default "raw")
